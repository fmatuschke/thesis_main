\documentclass[]{standalone}
\usepackage{pgfplots}
\usepackage{pgfplotstable}
\usepgfplotslibrary{groupplots}
\usepgfplotslibrary{polar}
\pgfplotsset{compat=1.16}

% https://tex.stackexchange.com/questions/267085/how-to-make-a-wind-rose-using-pgfplots-tikz?rq=1
% http://pgfplots.net/tikz/examples/color-maps/

\begin{document}
\makeatletter
\pgfplotsset{
    polar bar/.style={
        scatter,
        draw=none,
        mark=none,
        % point meta min=0,
        visualization depends on=rawy\as\rawy,
        /pgfplots/scatter/@post marker code/.add code={}{
            \pgfmathveclen{\pgf@x}{\pgf@y}
            \edef\radius{\pgfmathresult}
            \ifdim \pgfkeysvalueof{/data point/y} pt > 0pt
            \fill[fill=mapped color] 
                (\pgfkeysvalueof{/data point/x},-\pgfkeysvalueof{/data point/y})
                ++({\pgfkeysvalueof{/data point/x}-#1/2},\pgfkeysvalueof{/data point/y})
                arc [start angle=\pgfkeysvalueof{/data point/x}-#1/2,
                    delta angle=#1,
                    radius={\radius pt}
                ]
                -- +({\pgfkeysvalueof{/data point/x}+#1/2},-\rawy)
                arc [start angle=\pgfkeysvalueof{/data point/x}+#1/2,
                    delta angle=-#1,
                    radius={
                        (\pgfkeysvalueof{/data point/y} - \rawy) / (\pgfkeysvalueof{/data point/y} + 0.00001) * \radius pt
                    }
                ]
                --cycle;
            \fi
        }
    },
    polar bar/.default=30
}

\end{tikzpicture}
}
